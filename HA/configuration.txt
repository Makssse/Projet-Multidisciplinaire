template:
  - sensor:
      #le trieur qui répartie dans les 3 catégories n0, n1, n2
      - name: "Chrono CO2 salle1"
        unique_id: chrono_co2_salle1
        state: >
          {% set co2 = states('sensor.pico_salle1_co2') | float(-1) %}
          {% if co2 == 0 %}
            inconnu
          {% elif co2 <= 800 %}
            n0
          {% elif co2 <= 1500 %}
            n1
          {% else %}
            n2
          {% endif %}


      - name: "Indice ICONE Salle1"
        unit_of_measurement: "Niveau"
        state: >
          {% set t0 = states('sensor.temps_n0_personnalise') | float(0) %}
          {% set t1 = states('sensor.temps_n1_personnalise') | float(0) %}
          {% set t2 = states('sensor.temps_n2_personnalise') | float(0) %}
          {% set total = t0 + t1 + t2 %}

          {% if total > 0 %}
            {% set f1 = t1 / total %}
            {% set f2 = t2 / total %}
            {% set icone = 8.3 * log(1 + f1 + 3 * f2, 10) %}
            {{ icone | round(1) }}
          {% else %}
            0
          {% endif %}
